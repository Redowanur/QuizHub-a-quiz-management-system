300    /**
301     * Answer a new instance of this class with the specified opcode and name. Assume no rewrite.
302     *
303     * @param opcode int corresponding to the opcode's value
304     * @param name String printable name of the opcode
305     */
306    public ByteCodeForm(final int opcode, final String name) {
307        this(opcode, name, new int[] {opcode});
308    }
309
310    /**
311     * Answer a new instance of this class with the specified opcode, name, operandType and rewrite
312     *
313     * @param opcode int corresponding to the opcode's value
314     * @param name String printable name of the opcode
315     * @param rewrite int[] Array of ints. Operand positions (which will later be rewritten in ByteCodes) are indicated
316     *        by -1.
317     */
318    public ByteCodeForm(final int opcode, final String name, final int[] rewrite) {
319        this.opcode = opcode;
320        this.name = name;
321        this.rewrite = rewrite;
322        calculateOperandPosition();
323    }
324
325    protected void calculateOperandPosition() {
326        firstOperandIndex = -1;
327        operandLength = -1;
328
329        // Find the first negative number in the rewrite array
330        int iterationIndex = 0;
331        while (iterationIndex < rewrite.length) {
332            if (rewrite[iterationIndex] < 0) {
333                // Found the first opcode to substitute
334                firstOperandIndex = iterationIndex;
335                iterationIndex = rewrite.length;
336            } else {
337                iterationIndex++;
338            }
339        }
340
341        if (firstOperandIndex == -1) {
342            // Nothing more to do since the opcode has no operands
343            return;
344        }
345
346        // Find the last negative number in the rewrite array
347        int lastOperandIndex = -1;
348        iterationIndex = firstOperandIndex;
349        while (iterationIndex < rewrite.length) {
350            if (rewrite[iterationIndex] < 0) {
351                lastOperandIndex = iterationIndex;
352            }
353            iterationIndex++;
354        }
355
356        // Now we have the first index and the last index.
357        final int difference = lastOperandIndex - firstOperandIndex;
358
359        // If last < first, something is wrong.
360        if (difference < 0) {
361            throw new Error("Logic error: not finding rewrite operands correctly");
362        }
363        operandLength = difference + 1;
364    }

406     * This method will answer true if the receiver is a multi-bytecode instruction (such as aload0_putfield_super);
407     * otherwise, it will answer false.
408     *
409     * @return boolean true if multibytecode, false otherwise
410     */
411    public boolean hasMultipleByteCodes() {
412        // Currently, all multi-bytecode instructions
413        // begin with aload_0, so this is how we test.
414        if ((rewrite.length > 1) && (rewrite[0] == 42)) {
415            // If there's an instruction (not a negative
416            // number, which is an operand) after the
417            // aload_0, it's a multibytecode instruction.
418            return (rewrite[1] > 0);
419        }
420        return false;
421    }
